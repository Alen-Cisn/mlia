# Códigos de ejemplo de MLia

```
    decl variable = 20
    decl otraVariable = 30
    decl miFunción arg1 arg2 =
        + arg1 arg2

    decl +función-con/símbolos!raros arg1 =
        0


    while < variable otraVariable do
        otraVariable = - otraVariable 1;

        decl nuevoValor =
            match otraVariable with
            | 30 -> 
                decl a = 2 in
                miFunción a 3
            | 20 -> 
                2 in

        print nuevoValor
    done

    1
```
